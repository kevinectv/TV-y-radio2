    <script>
      function setDarkModeMenu() {
        if ("theme" in localStorage) {
          $('#darkmodeauto').removeClass("disabled");
        }
        else {
          $('#darkmodeauto').addClass("disabled");
        }
      }
      function initDarkMode() {
        if ( localStorage.theme === "dark" || (!("theme" in localStorage) && window.matchMedia("(prefers-color-scheme: dark)").matches) ) {
          document.documentElement.classList.add("dark");
        } else {
          document.documentElement.classList.remove("dark");
        }
        setDarkModeMenu()
      }

      //initDarkMode()
      window.matchMedia('(prefers-color-scheme: dark)').addListener(initDarkMode);

      let toggleDark = () => {
        let result = document.documentElement.classList.toggle("dark");
        localStorage.theme = result ? "dark" : "light";
        setDarkModeMenu()
      };

      let resetDark = () => {
        localStorage.removeItem("theme");
        initDarkMode()
      };

      $('#darkmode').click(function(){ toggleDark(); return true; });
      $('#darkmodeauto').click(function(){ resetDark(); return true; });

      function initMenuDropdowns() {
        const isMobile = window.isMdblistMobile();
        const $dropdowns = $('.ui.jsimple.dropdown');
        $dropdowns.dropdown('destroy');
        if (isMobile) {
          $dropdowns.addClass('upward');
        } else {
          $dropdowns.removeClass('upward');
        }
        $dropdowns.dropdown(isMobile ? { on: 'click', direction: 'upward' } : {});
      }

      $(function() {
        let lastIsMobile = window.isMdblistMobile();
        initMenuDropdowns();
        $(window).on('resize', function() {
          const nowIsMobile = window.isMdblistMobile();
          if (nowIsMobile !== lastIsMobile) {
            lastIsMobile = nowIsMobile;
            initMenuDropdowns();
          }
        });

        
      });

</script>

<script>
  // Shared helper for resetting mdblist-only Arr status (deleted -> clear).
  // This does not call Radarr/Sonarr; it only clears TraktWatched.radarr/sonarr locally.
  window.resetArrStatus = function resetArrStatus(opts) {
    opts = opts || {};
    var arr = opts.arr;
    var tmdbid = opts.tmdbid;
    var tvdbid = opts.tvdbid;
    var csrf = $('meta[name="csrf-token"]').attr('content') || 'yo1fUsiUzualQxAyiKRhfajfgoiBIue5fldRQjRqL5j6dDFxrElkOJ5RfgDvFGbS';

    if (!arr || (arr !== 'radarr' && arr !== 'sonarr')) {
      return $.Deferred().reject('invalid-arr').promise();
    }

    return $.ajax({
      url: '/ajax/arr_status/reset/',
      type: 'POST',
      headers: { 'X-CSRFToken': csrf },
      contentType: 'application/json',
      dataType: 'json',
      data: JSON.stringify({
        arr: arr,
        tmdbid: tmdbid,
        tvdbid: tvdbid
      })
    });
  };

  // Delegated handler for all reset-arr-status dropdown menu items (mdblistarr & notifiarr).
  $(document).on('click', '[data-action="reset-arr-status"]', function() {
    var $item = $(this);
    var $dropdown = $item.closest('.ui.dropdown');
    var iconElement = $dropdown.find('.icon');
    var tmdbid = $dropdown.data('tmdbid') || iconElement.data('tmdbid');
    var tvdbid = $dropdown.data('tvdbid') || iconElement.data('tvdbid');
    var arr = $item.data('arr') || (tmdbid ? 'radarr' : 'sonarr');
    var client = $dropdown.hasClass('notifiarr') ? 'Notifiarr' : 'MDBListarr';

    window.resetArrStatus({ arr: arr, tmdbid: tmdbid, tvdbid: tvdbid })
      .done(function(resp) {
        if (resp && resp.response === 200) {
          $('body').toast({ class: 'success', message: 'Status reset', showProgress: 'bottom', displayTime: 4000 });
          var $img = iconElement.find('img').first();
          var $dot = $dropdown.find('.lib-arr-dot');
          var arrLabel = arr === 'radarr' ? 'Radarr' : 'Sonarr';
          $dot.removeClass('arr-s-downloaded arr-s-missing arr-s-deleted arr-s-excluded').addClass('arr-s-none');
          $dropdown.find('.menu > .header').first().text(arrLabel + ': Not added');
          $item.prev('.divider').remove();
          $item.remove();
          if (typeof refreshLibArrTooltip === 'function') {
            refreshLibArrTooltip($dropdown);
          }
          if (arr === 'radarr') {
            $img.attr('src', '/static/radarr-o.png');
            iconElement.attr('data-html', 'Add Movie to Radarr via ' + client);
          } else {
            $img.attr('src', '/static/sonarr-o.png');
            iconElement.attr('data-html', 'Add Show to Sonarr via ' + client);
          }
        } else {
          $('body').toast({ class: 'error', message: 'Unable to reset status', showProgress: 'bottom', displayTime: 5000 });
        }
      })
      .fail(function() {
        $('body').toast({ class: 'error', message: 'Unable to reset status', showProgress: 'bottom', displayTime: 5000 });
      });
  });
</script>
